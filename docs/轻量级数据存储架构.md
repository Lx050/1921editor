# 轻量级数据存储架构设计

## 需求明确

1. **飞书多维表格存储元数据**：
   - 标题
   - 编辑者
   - 图文来源
   - 文案作者
   - 创建时间
   - 内容ID
   - 图片链接（微信返回的URL）

2. **服务器不存储图片**：
   - 只存储元数据（JSON文件或轻量数据库）
   - 图片直接从微信获取URL

3. **参考微信API模式**：
   - 使用微信API上传图片获取URL
   - 保存URL而不是图片文件

### 第二步：轻量后端实现

```typescript
// src/backend/dataManager.ts
import fs from 'fs/promises';
import path from 'path';

class DataManager {
  private dataDir = path.join(__dirname, '../data');
  private metadataFile = path.join(this.dataDir, 'metadata.json');
  private indexFile = path.join(this.dataDir, 'index.json');

  constructor() {
    this.ensureDataDir();
  }

  // 确保数据目录存在
  private async ensureDataDir() {
    try {
      await fs.mkdir(this.dataDir, { recursive: true });
    } catch (error) {
      console.error('创建数据目录失败:', error);
    }
  }

  // 保存元数据
  async saveMetadata(metadata: ContentMetadata): Promise<void> {
    try {
      // 读取现有数据
      const existingData = await this.loadAllMetadata();

      // 添加或更新
      existingData[metadata.id] = {
        ...metadata,
        updatedAt: new Date().toISOString()
      };

      // 保存到文件
      await fs.writeFile(
        this.metadataFile,
        JSON.stringify(existingData, null, 2),
        'utf8'
      );

      // 更新索引（用于快速查询）
      await this.updateIndex(metadata);

      console.log(`[DataManager] 元数据保存成功: ${metadata.id}`);
    } catch (error) {
      console.error('[DataManager] 保存元数据失败:', error);
      throw error;
    }
  }

  // 加载所有元数据
  async loadAllMetadata(): Promise<Record<string, ContentMetadata>> {
    try {
      const data = await fs.readFile(this.metadataFile, 'utf8');
      return JSON.parse(data);
    } catch (error) {
      if ((error as NodeJS.ErrnoException).code === 'ENOENT') {
        return {};
      }
      throw error;
    }
  }

  // 获取单个元数据
  async getMetadata(id: string): Promise<ContentMetadata | null> {
    const allData = await this.loadAllMetadata();
    return allData[id] || null;
  }

  // 更新索引（用于飞书同步）
  private async updateIndex(metadata: ContentMetadata): Promise<void> {
    try {
      let index: any[] = [];

      try {
        const indexData = await fs.readFile(this.indexFile, 'utf8');
        index = JSON.parse(indexData);
      } catch {
        // 文件不存在，使用空数组
      }

      // 查找并更新或添加
      const existingIndex = index.findIndex(item => item.id === metadata.id);
      const indexItem = {
        id: metadata.id,
        title: metadata.title,
        editor: metadata.editor,
        source: metadata.source,
        author: metadata.author,
        category: metadata.category,
        createdAt: metadata.createdAt,
        updatedAt: metadata.updatedAt,
        status: metadata.status,
        imageCount: metadata.wechatImages.length,
        summary: metadata.summary,
        // 飞书需要的URL
        detailUrl: `https://your-domain.com/content/${metadata.id}`
      };

      if (existingIndex >= 0) {
        index[existingIndex] = indexItem;
      } else {
        index.push(indexItem);
      }

      // 按创建时间倒序排列
      index.sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());

      await fs.writeFile(
        this.indexFile,
        JSON.stringify(index, null, 2),
        'utf8'
      );
    } catch (error) {
      console.error('[DataManager] 更新索引失败:', error);
    }
  }

  // 获取索引数据（用于飞书同步）
  async getIndex(): Promise<any[]> {
    try {
      const data = await fs.readFile(this.indexFile, 'utf8');
      return JSON.parse(data);
    } catch (error) {
      return [];
    }
  }

  // 查询接口
  async query(filter?: {
    category?: string;
    editor?: string;
    status?: string;
    startDate?: string;
    endDate?: string;
    limit?: number;
    offset?: number;
  }): Promise<{ items: ContentMetadata[], total: number }> {
    const allData = await this.loadAllMetadata();
    let items = Object.values(allData);

    // 应用筛选
    if (filter) {
      if (filter.category) {
        items = items.filter(item => item.category === filter.category);
      }
      if (filter.editor) {
        items = items.filter(item => item.editor === filter.editor);
      }
      if (filter.status) {
        items = items.filter(item => item.status === filter.status);
      }
      if (filter.startDate) {
        items = items.filter(item => item.createdAt >= filter.startDate);
      }
      if (filter.endDate) {
        items = items.filter(item => item.createdAt <= filter.endDate);
      }
    }

    // 排序
    items.sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());

    // 分页
    const total = items.length;
    if (filter?.limit) {
      const offset = filter.offset || 0;
      items = items.slice(offset, offset + filter.limit);
    }

    return { items, total };
  }
}

export const dataManager = new DataManager();
```

```typescript
// src/backend/webhook.ts
import express from 'express';
import { dataManager } from './dataManager';

const app = express();
app.use(express.json());

// Webhook接收端点
app.post('/webhook/content', async (req, res) => {
  try {
    const { event, data } = req.body;

    switch (event) {
      case 'content_created':
        await handleContentCreated(data);
        break;

      case 'content_updated':
        await handleContentUpdated(data);
        break;

      case 'content_published':
        await handleContentPublished(data);
        break;

      case 'images_uploaded':
        await handleImagesUploaded(data);
        break;

      default:
        console.log('[Webhook] 未知事件:', event);
    }

    res.json({ success: true });
  } catch (error) {
    console.error('[Webhook] 处理失败:', error);
    res.status(500).json({ error: error.message });
  }
});

// 处理内容创建
async function handleContentCreated(data: any) {
  const metadata: ContentMetadata = {
    id: data.id,
    title: data.title,
    editor: data.editor,
    source: data.source,
    author: data.author,
    category: data.category || 'default',
    tags: data.tags || [],
    status: 'draft',
    wechatImages: data.wechatImages || [],
    createdAt: new Date().toISOString(),
    updatedAt: new Date().toISOString(),
    viewCount: 0,
    likeCount: 0,
    summary: data.summary || generateSummary(data.content),
    storagePath: data.storagePath || `contents/${data.id}.json`
  };

  await dataManager.saveMetadata(metadata);

  // 触发飞书同步
  await triggerFeishuSync(metadata);
}

// 处理图片上传
async function handleImagesUploaded(data: { contentId: string, images: WechatImage[] }) {
  const metadata = await dataManager.getMetadata(data.contentId);
  if (!metadata) {
    throw new Error(`内容不存在: ${data.contentId}`);
  }

  // 更新图片信息
  metadata.wechatImages = data.images;
  metadata.updatedAt = new Date().toISOString();

  await dataManager.saveMetadata(metadata);
}

// 生成内容摘要
function generateSummary(content: string): string {
  // 移除HTML标签
  const text = content.replace(/<[^>]+>/g, '');
  // 返回前200字符
  return text.substring(0, 200) + (text.length > 200 ? '...' : '');
}

}
```

### 第四步：前端集成

```typescript
// src/utils/contentManager.ts
import { uploadImages } from './wechatApi';
import type { WechatImage } from '@/types';

export class ContentManager {
  private webhookUrl: string;

  constructor() {
    this.webhookUrl = process.env.VITE_WEBHOOK_URL || 'http://localhost:3001/webhook/content';
  }

  // 保存内容草稿
  async saveDraft(contentData: {
    title: string;
    content: string;
    editor: string;
    source: string;
    author: string;
    category?: string;
    tags?: string[];
    images?: File[];
  }) {
    // 1. 生成唯一ID
    const id = `content_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;

    // 2. 处理图片上传
    let wechatImages: WechatImage[] = [];
    if (contentData.images && contentData.images.length > 0) {
      wechatImages = await uploadImages(contentData.images);
    }

    // 3. 发送到Webhook
    const webhookData = {
      event: 'content_created',
      data: {
        id,
        ...contentData,
        wechatImages,
        content: contentData.content
      }
    };

    await fetch(this.webhookUrl, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(webhookData)
    });

    return { id, success: true };
  }

  // 更新内容
  async updateContent(id: string, updates: Partial<any>) {
    const webhookData = {
      event: 'content_updated',
      data: { id, ...updates }
    };

    await fetch(this.webhookUrl, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(webhookData)
    });
  }

  // 发布内容
  async publishContent(id: string) {
    // 1. 获取内容数据
    const metadata = await this.getMetadata(id);
    if (!metadata) {
      throw new Error('内容不存在');
    }

    // 2. 创建微信草稿
    const draftArticle = {
      title: metadata.title,
      author: metadata.author,
      digest: metadata.summary,
      content: metadata.content,
      content_source_url: `https://your-domain.com/content/${id}`,
      thumb_media_id: metadata.wechatImages[0]?.mediaId,
      show_cover_pic: 1,
      need_open_comment: 1,
      only_fans_can_comment: 0
    };

    // 调用微信API创建草稿
    const { createDraft } = await import('./wechatApi');
    const draftResponse = await createDraft(draftArticle);

    // 3. 更新状态
    await this.updateContent(id, {
      status: 'published',
      wechatDraftId: draftResponse.draft_id,
      publishedAt: new Date().toISOString()
    });

    // 4. 发送发布事件
    await fetch(this.webhookUrl, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        event: 'content_published',
        data: { id, draftId: draftResponse.draft_id }
      })
    });
  }

  // 获取元数据
  async getMetadata(id: string) {
    const response = await fetch(`/api/metadata/${id}`);
    return response.json();
  }
}
```

```typescript
// 飞书插件 - 记录接口
export async function handleRecords(req: Request) {
  const { maxPageSize, pageToken } = JSON.parse(req.body.params);

  // 从后端获取数据
  const response = await fetch('http://localhost:3001/api/index');
  const index = await response.json();

  // 分页处理
  const offset = pageToken ? parseInt(pageToken) : 0;
  const records = index.slice(offset, offset + maxPageSize);

  return {
    code: 0,
    data: {
      hasMore: offset + maxPageSize < index.length,
      nextPageToken: offset + maxPageSize < index.length ? (offset + maxPageSize).toString() : null,
      records: records.map(item => ({
        primaryID: item.id,
        data: {
          id: item.id,
          title: item.title,
          editor: item.editor,
          source: item.source,
          author: item.author,
          category: item.category,
          status: item.status,
          createdAt: new Date(item.createdAt).getTime(),
          updatedAt: new Date(item.updatedAt).getTime(),
          imageCount: item.imageCount,
          viewCount: item.viewCount || 0,
          summary: item.summary,
          detailUrl: item.detailUrl
        }
      }))
    }
  };
}
```
